18  * This is a C function.
19  * To include within a C++ program use
20  * #include "mdaio.h"

22  *
23  * Example usage: See the transpose_array function at the bottom of this file
24  * (and note that this would be more elegant with C++) ;)
25  *
26  * Another example usage: scda_ss/processing/filters/expfilter/expfilter.c
27  *
28  * The key trick here is the convenience of the functions
29  * mda_read_float32, mda_read_byte, ..., and mda_write_float, mda_write_byte, ...
30  * because they allows you to read/write of a chosen type without worrying about
31  * the actual data storage type. Normally you would need to have if/else statements
32  * to handle every possible type. Think about how horrible this would be for even
33  * the simple transpose_array function below.
34  *
35  * Note that we don't handle the MDA_TYPE_COMPLEX case in this file
36  *
37  * */

54 struct MDAIO_HEADER {
